CREATE TABLE USER(

userID VARCHAR(50) NOT NULL,
userPassword VARCHAR(50) NOT NULL,
userName VARCHAR(50) NOT NULL,
userGender VARCHAR(10) NOT NULL,
userSchoolName VARCHAR(20) NOT NULL,
userSchool VARCHAR(50) NOT NULL,
userGrade VARCHAR(50) NOT NULL,
userClassroom VARCHAR(10) NOT NULL,
userNumber VARCHAR(10) NOT NULL,

CONSTRAINT PRIMARY KEY(userID)

);


CREATE TABLE RECORDDATA(

userID VARCHAR(50) NOT NULL,
recordDate VARCHAR(100) NOT NULL,
recordMeter VARCHAR(50) NOT NULL,
trackCount VARCHAR(50) NOT NULL,
trackTimeDate VARCHAR(300) NOT NULL,
allTrackTimeDate VARCHAR(100) NOT NULL

);
